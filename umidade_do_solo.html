int sensorPin = A0;
int valorSensor = 0;

void setup() {
Serial.begin(9600); // Inicializa a comunicação serial
}

void loop() {
valorSensor = analogRead(sensorPin); // Lê o valor do sensor
Serial.print("Umidade do solo: ");
Serial.println(valorSensor); // Exibe o valor no monitor serial

delay(1000); // Aguarda 1 segundo antes de ler novamente
}
