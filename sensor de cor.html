const int s0 = 8;
const int s1 = 9;
const int s2 = 12;
const int s3 = 11;
const int out = 10;

void setup() {
  pinMode(s0, OUTPUT);
  pinMode(s1, OUTPUT);
  pinMode(s2, OUTPUT);
  pinMode(s3, OUTPUT);
  pinMode(out, INPUT);
 
  digitalWrite(s0, HIGH);
  digitalWrite(s1, LOW);
 
  Serial.begin(9600);
}

void loop() {
  int red = readColor(LOW, LOW);   // Ler vermelho
  int green = readColor(HIGH, HIGH); // Ler verde
  int blue = readColor(LOW, HIGH);  // Ler azul
 
  String color = determineColor(red, green, blue);
 
  Serial.print("Red: ");
  Serial.print(red);
  Serial.print(" Green: ");
  Serial.print(green);
  Serial.print(" Blue: ");
  Serial.print(blue);
 
  Serial.print(" Cor detectada: ");
  Serial.println(color);
 
  delay(1000);
}

int readColor(int s2State, int s3State) {
  digitalWrite(s2, s2State);
  digitalWrite(s3, s3State);
  delay(50); // Pequeno atraso para estabilizar a leitura
  return pulseIn(out, LOW);
}
//Área de programação

String determineColor(int red, int green, int blue) {
  if (red < 50 && green < 50 && blue < 50) {
    return "Branco";
  } else if (red > 200 && green > 200 && blue > 200) {
    return "Preto";
  } else if (green < red && green < red && green < 200 && blue < 200 ) {
    return "Verde";
  } else {
    return "Outra cor";
  }
}
