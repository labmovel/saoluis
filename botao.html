int botao = 2; // Pino onde o botão está conectado

void setup() {
  pinMode(botao, INPUT_PULLUP); // Usa resistor de pull-up interno
  Serial.begin(9600);           // Inicia a comunicação serial
}

void loop() {
  int estadoBotao = digitalRead(botao);

  if (estadoBotao == LOW) {
    Serial.println("Botão pressionado");
  } else {
    Serial.println("Botão solto");
  }
  delay(500); // Aguarda meio segundo para evitar muitas mensagens
}
