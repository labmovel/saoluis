float distancia;

void setup()
{
pinMode (13, OUTPUT); //trig
pinMode (12, INPUT); //echo
Serial.begin (9600);
}

void loop()

{
digitalWrite(13, LOW);
delay(0005);
digitalWrite(13, HIGH);
delay(0010);
digitalWrite(13, LOW);


distancia = pulseIn (12, HIGH);
distancia = distancia/58;
Serial.println (distancia);
}
