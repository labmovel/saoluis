// Pinos do motor
int M1IN1 = 4;
int M1IN2 = 5;
int ENA = 3; // Controle de velocidade (PWM)

void setup() {
  pinMode(M1IN1, OUTPUT);
  pinMode(M1IN2, OUTPUT);
  pinMode(ENA, OUTPUT);

  // Define velocidade constante (ex.: intermediária)
  analogWrite(ENA, 150);
}

void loop() {
  // Sentido horário por 1 segundo
  digitalWrite(M1IN1, HIGH);
  digitalWrite(M1IN2, LOW);
  delay(1000);

  // Sentido anti-horário por 1 segundo
  digitalWrite(M1IN1, LOW);
  digitalWrite(M1IN2, HIGH);
  delay(1000);
}


Com voids e diferentes controles de velocidade

// Pinos do motor
int M1IN1 = 4;
int M1IN2 = 5;
int ENA = 3; // Controle de velocidade (PWM)

// Função para girar para frente com velocidade alta
void frente() {
  analogWrite(ENA, 230); // Alta velocidade
  digitalWrite(M1IN1, HIGH);
  digitalWrite(M1IN2, LOW);
}

// Função para girar para trás com velocidade baixa
void tras() {
  analogWrite(ENA, 100); // Baixa velocidade
  digitalWrite(M1IN1, LOW);
  digitalWrite(M1IN2, HIGH);
}

// Função para parar o motor
void parar() {
  digitalWrite(M1IN1, LOW);
  digitalWrite(M1IN2, LOW);
}

void setup() {
  pinMode(M1IN1, OUTPUT);
  pinMode(M1IN2, OUTPUT);
  pinMode(ENA, OUTPUT);
}

void loop() {
  frente();   // Frente com velocidade alta
  delay(2000);

  tras();     // Trás com velocidade baixa
  delay(2000);

  parar();    // Para o motor
  delay(1000);
}
