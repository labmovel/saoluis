#include <SoftwareSerial.h>
 
// Definição dos Pinos nas portas do Arduino
int M1IN1 = 4;
int M1IN2 = 5;
 
void setup() {
  // Definição dos Pinos como saída
  pinMode(M1IN1, OUTPUT);
  pinMode(M1IN2, OUTPUT);
}
 
void loop() {
  // O código abaixo é para girar no sentido horário por 1 segundo
  digitalWrite(M1IN1, HIGH);
  digitalWrite(M1IN2, LOW);
  delay(1000);
 
  // O código abaixo é para girar no sentido anti-horário por 1 segundo
  digitalWrite(M1IN1, LOW);
  digitalWrite(M1IN2, HIGH);
  delay(1000);
}


OUTRAS POSSIBILIDADES




#include <SoftwareSerial.h>
 
// Definição dos Pinos nas portas do Arduino
int M1IN1 = 4;
int M1IN2 = 5;
int M2IN3 = 6;
int M2IN4 = 7;
 
void setup() {
  // Definição dos Pinos como saída
  pinMode(M1IN1, OUTPUT);
  pinMode(M1IN2, OUTPUT);
  pinMode(M2IN3, OUTPUT);
  pinMode(M2IN4, OUTPUT);
}
void loop() {
  // O código abaixo faz os 2 motores se movimentarem no mesmo sentido. Exemplo: frente
  digitalWrite(M1IN1, LOW);
  digitalWrite(M1IN2, HIGH);
  digitalWrite(M2IN3, LOW);
  digitalWrite(M2IN4, HIGH);
}
 
A combinação ou troca de LOW por HIGH e vice-versa define as direções ou a paralisação dos motores. Abaixo alguns exemplos que podem ser implementados com o código:

Para frente

digitalWrite(M1IN1, LOW);
digitalWrite(M1IN2, HIGH);
digitalWrite(M2IN3, LOW);
digitalWrite(M2IN4, HIGH);
 

Para trás

digitalWrite(M1IN1, HIGH);
digitalWrite(M1IN2, LOW);
digitalWrite(M2IN3, HIGH);
digitalWrite(M2IN4, LOW);
 

Para direita

digitalWrite(M1IN1, HIGH);
digitalWrite(M1IN2, LOW);
digitalWrite(M2IN3, LOW);
digitalWrite(M2IN4, HIGH);
 

Para esquerda

digitalWrite(M1IN1, LOW);
digitalWrite(M1IN2, HIGH);
digitalWrite(M2IN3, HIGH);
digitalWrite(M2IN4, LOW);
 

Parar motores

digitalWrite(M1IN1, LOW);
digitalWrite(M1IN2, LOW);
digitalWrite(M2IN3, LOW);
digitalWrite(M2IN4, LOW);
 

Uma outra possibilidade, que otimiza boa parte do código, principalmente quando você pensa, por exemplo, em programar um robô para ir de um lugar até outro (Ex: Frente, Direita, Frente, Esquerda, Pare), é utilizando funções (voids). O código de exemplo abaixo é para 2 motores:

#include <SoftwareSerial.h>
 
// Definição dos Pinos nas portas do Arduino
int M1IN1 = 4;
int M1IN2 = 5;
int M2IN3 = 6;
int M2IN4 = 7;
 
void setup() {
  // Definição dos Pinos como saída
  pinMode(M1IN1, OUTPUT);
  pinMode(M1IN2, OUTPUT);
  pinMode(M2IN3, OUTPUT);
  pinMode(M2IN4, OUTPUT);
}
 
//
Dentro do loop você só vai chamar as funções definidas no void (Ex:
frente, esquerda, parar). No exemplo, fazemos o robô se movimentar para
frente por 3 segundos. Ele fica parado 1 segundo e depois move para
trás por 3 segundos.
void loop() {
  frente();
  delay(3000);
  parar();
  delay(1000);
  tras();
  delay(3000);
}
 
// Aqui você define um void para cada função que o motor irá realizar:
void frente() {
  digitalWrite(M1IN1, LOW);
  digitalWrite(M1IN2, HIGH);
  digitalWrite(M2IN3, LOW);
  digitalWrite(M2IN4, HIGH);
}
 
void tras() {
  digitalWrite(M1IN1, HIGH);
  digitalWrite(M1IN2, LOW);
  digitalWrite(M2IN3, HIGH);
  digitalWrite(M2IN4, LOW);
}
 
void direita() {
  digitalWrite(M1IN1, HIGH);
  digitalWrite(M1IN2, LOW);
  digitalWrite(M2IN3, LOW);
  digitalWrite(M2IN4, HIGH);
}
 
void esquerda() {
  digitalWrite(M1IN1, LOW);
  digitalWrite(M1IN2, HIGH);
  digitalWrite(M2IN3, HIGH);
  digitalWrite(M2IN4, LOW);
}
 
void parar() {
digitalWrite(M1IN1, LOW);
digitalWrite(M1IN2, LOW);
digitalWrite(M2IN3, LOW);
digitalWrite(M2IN4, LOW);
}



CONTROLE DE VELOCIDADE

Antes do setup
int ENA = 3;
int ENB = 11

No setup
pinMode(ENA,OUTPUT);
pinMode(ENB,OUTPUT);

Exemplos de velocidade:
//Alta
analogWrite(ENA,230);
analogWrite(ENB,230);

//Intermediaria
analogWrite(ENA,150);
analogWrite(ENB,150);

//Baixa
analogWrite(ENA,80);
analogWrite(ENB,80);
